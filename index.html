Organizador de archivos:

# Python 3
import os
carpeta = "C:/Descargas"
for archivo in os.listdir(carpeta):
    ruta = os.path.join(carpeta, archivo)
    if os.path.isfile(ruta):
        ext = archivo.split(".")[-1]
        carpeta_ext = os.path.join(carpeta, ext)
        os.makedirs(carpeta_ext, exist_ok=True)
        os.rename(ruta, os.path.join(carpeta_ext, archivo))

Convertidor de PDFs a imágenes:

# Python 3
from pdf2image import convert_from_path
pages = convert_from_path('documento.pdf',300)
for i, page in enumerate(pages):
    page.save(f'pagina_{i+1}.jpg','JPEG')
